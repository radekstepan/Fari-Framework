172:     function getPost($key=NULL, $filter='text') {
173:         // can we apply the filter passed?
174:         try {
175:             if (!method_exists('Fari_Escape', $filter)) {
176:                 // ... throw exception if filter function is invalid
177:                 throw new Fari_Exception('Fari_Escape::' . $filter . ' is not a valid escaping function.');
178:             }
179:         } catch (Fari_Exception $exception) { $exception->fire(); }
180: 
181:         // return the value(s), filtered
182:         if (isset($key)) {
183:             return ($this->isAjax()) ?
184:                 // decode first in JavaScript context
185:                 Fari_Escape::$filter(Fari_Decode::javascript($this->post->$key)) :
186:                 Fari_Escape::$filter($this->post->$key);
187:         } else {
188:             // get the values
189:             $post = $this->post->values;
190: 
191:             // decode from AJAX?
192:             if ($this->isAjax()) $post = Fari_Decode::javascript($post);
193: 
194:             // filter them
195:             foreach ($post as $key => &$value) $value = Fari_Escape::$filter($value);

197:             return $post;
198:         }

 22:     /**#@+ filename suffix */
 23:     const VIEW_SUFFIX = '.phtml';
 24:     /**#@-*/

 26:     /** @var values from Fari_Bag to extract */
 27:     private $values = NULL;
 28: 
 29:     /**
 30:      * The constructor extracts the bag of values passed from the Presenter.
 31:      * @param A bag of key:value symbols
 32:      */
 33:     public function __construct(array $values) {
 34:         $this->values = $values;
 35: 
 36:         // include view helpers
 37:         include BASEPATH . "/fari/Application/ApplicationViewHelpers" . EXT;
 38:         // set the values in the helpers so they 'catch' them
 39:         Fari_ApplicationViewHelper::setValues($this->values);
 40:     }

 42:     /**
 43:      * Displays the view after (optionally) render function in a presenter has been executed.
 44:      * @param string $viewName View name to display
 45:      * @param string $contentType Specifies optional content type for the view
 46:      * @param string $extraParam Extra parameter to find the view by
 47:      */
 48:     public function render($viewName) {
 49:         assert('strlen($viewName) > 0; // view name cannot be empty');
 50: 
 51:         // form views path
 52:         $path = BASEPATH . '/' . APP_DIR . '/views/';
 53:         // create template file path
 54:         $viewFile = $path . $viewName . self::VIEW_SUFFIX;
 55:         
 56:         // check if view exists
 57:         $this->isViewValid($viewFile);
 58: 
 59:         // are we using a @layout file?
 60:         $temp = explode('/', $viewName);
 61:         assert('count($temp) == 2; // $viewName needs to consist of "presenter/file"');
 62: 
 63:         // define presenter name for easy access
 64:         Fari_ApplicationViewHelper::setPresenter($temp[0]);
 65: 
 66:         // custom layout named after our presenter
 67:         if (file_exists($layout = $path . '@' . strtolower($temp[0]) . self::VIEW_SUFFIX)) {
 68:             $this->includeLayoutAndView($layout, $viewFile);
 69:         // application level layout
 70:         } else if (file_exists($layout = $path . '@application' . self::VIEW_SUFFIX)) {
 71:             $this->includeLayoutAndView($layout, $viewFile);
 72:         } else {
 73:             // import key:value array into symbol table
 74:             extract($this->values, EXTR_SKIP);
 75:             
 76:             // no soup for you!
 77:             $devMode = Fari_ApplicationEnvironment::isDevelopment();
 78:             if ($devMode) echo "\n<!-- begin {$viewFile} -->\n";
 79:             include $viewFile;
 80:             if ($devMode) echo "\n<!-- end {$viewFile} -->\n";
 81:         }
 82:     }
 83:     
 84:     /**
 85:      * Check if view file can be included. Will throw Fari_Exception on error.
 86:      * @param string $viewFile to check for existence
 87:      * @throws Fari_Exception
 88:      * @return void
 89:      */
 90:     private function isViewValid($viewFile) {
 91:         try {
 92:             // check if file path exists
 93:             if (!file_exists($viewFile)) {
 94:                 throw new Fari_Exception('View not located in: ' . $viewFile);
 95:             }
 96:         } catch (Fari_Exception $exception) { $exception->fire(); }
 97:     }

 99:     /**
100:      * Include layout file and view file included in $template var.
101:      * @param string $layout file path
102:      * @param string $view file path
103:      */
104:     private function includeLayoutAndView($layout, $view) {
105:         // we are defining our own var $template so check if exists
106:         assert('!array_key_exists(\'template\', $this->values);
107:             // $template variable will be overwritten, do not use it');
108: 
109:         // import key:value array into symbol table
110:         extract($this->values, EXTR_SKIP);
111: 
112:         // display paths to files in development mode
113:         $devMode = Fari_ApplicationEnvironment::isDevelopment();
114: 
115:         ob_start();

117:         if ($devMode) echo "\n<!-- begin {$view} -->\n";
118:         include $view;
119:         if ($devMode) echo "\n<!-- end {$view} -->\n";
120:         $template = ob_get_contents();
121:         ob_end_clean();
122: 
123:         // call parent layout
124:         if ($devMode) echo "<!-- begin {$layout} -->\n";
125:         include $layout;
126:         if ($devMode) echo "\n<!-- end {$layout} -->\n";
127:     }

21: /**
22:  * Autoload Model classes when needed (before exception is thrown).
23:  * @param string $className
24:  */
25: function __autoload($className) {
26:     // are we working with a Fari Classes?
27:     if (strpos($className, 'Fari_') === FALSE) {
28:         // the only exception being a heavily used ORM
29:         if ($className == 'Table') {
30:             $classFilePath = BASEPATH . '/fari/Db/DbTable' . EXT;
31:         } else {
32:             // explode the class name by camel case
33:             $className = splitCamelCase($className);
34:             // the first name in a class could be a folder name, is it?
35:             if (is_dir(BASEPATH . '/'. APP_DIR . '/models/' . $className[0])) {
36:                 // it is, prefix the model with a directory name
37:                 $classFilePath = BASEPATH . '/'. APP_DIR . '/models/' . $className[0] . '/'
38:                 . implode('', $className) . EXT;
39:             } else {
40:                 // nah, the 'default'
41:                 $classFilePath = BASEPATH . '/'. APP_DIR . '/models/' . implode($className) . EXT;
42:             }
43:         }
44:     } else {
45:         // remove Fari_ and build path
46:         $className = splitCamelCase(substr($className, 5));
47:         $classFilePath = BASEPATH . '/fari/' . $className[0] . '/' . implode('', $className) . EXT;
48:     }
49: 
50:     try {
51:         // check that model class exists
52:         if (!file_exists($classFilePath)) {
53:             throw new Fari_Exception('Missing Class: ' . $classFilePath . '.');
54:         } else include($classFilePath); // include file
55:     } catch (Fari_Exception $exception) { $exception->fire(); }
56: }

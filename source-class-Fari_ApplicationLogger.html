 14: /**
 15:  * An observer saving information to log files.
 16:  * @example $ tail -f log/db.log
 17:  *
 18:  * @copyright Copyright (c) 2008, 2010 Radek Stepan
 19:  * @package   Fari Framework\Application
 20:  */
 21: final class Fari_ApplicationLogger {
 22: 
 23:     /**#@+ directory where to save logs */
 24:     const LOG_DIRECTORY = '/log';
 25:     /**#@-*/
 26: 
 27:     /** @var array of counters for each log file type */
 28:     private $counters = array();
 29: 
 30:     /**
 31:      * Constructs the logger checking if the logs directory is writable.
 32:      */
 33:     public function __construct() {
 34:         try {
 35:             // can we write into the dir?
 36:             if (!is_writable(BASEPATH . self::LOG_DIRECTORY)) {
 37:                 throw new Fari_Exception('Logs directory ' . self::LOG_DIRECTORY . ' is not writable.');
 38:             }
 39:         } catch (Fari_Exception $exception) { $exception->fire(); }
 40:     }
 41: 
 42:     /**
 43:      * Update this observer with subject values.
 44:      * @param Fari_ApplicationLoggerSubject $subject
 45:      */
 46:     public function update(Fari_ApplicationLoggerSubject $subject) {
 47:         // determine if type is set
 48:         $type = $subject->type();
 49: 
 50:         try { if (empty($type)) {
 51:             throw new Fari_Exception('You need to specify observer subject type.'); }
 52:         } catch (Fari_Exception $exception) { $exception->fire(); }
 53: 
 54:         // update the counters kept for each type of log so we can switch between colors etc.
 55:         $this->updateCounters($type);
 56: 
 57:         // write the information into a log file
 58:         $this->writeLog($type, $subject->value);
 59:     }
 60: 
 61:     /**
 62:      * Type of the log file we are saving to
 63:      * @param string $type
 64:      */
 65:     private function updateCounters($type) {
 66:         assert('!empty($type) && is_string($type); // log file type better be set!');
 67: 
 68:         if (array_key_exists($type, $this->counters)) {
 69:             // increase count
 70:             $this->counters[$type]++;
 71:         } else {
 72:             // create & start off count
 73:             $this->counters[$type] = 0;
 74:         }
 75:     }
 76: 
 77:     /**
 78:      * Write content to a log file.
 79:      * @param string $type
 80:      * @param string $content
 81:      */
 82:     private function writeLog($type, $content) {
 83:         try {
 84:             // is content a string?
 85:             if (!is_string($content)) {
 86:                 throw new Fari_Exception('Content needs to be in string format!');
 87:             }
 88:         } catch (Fari_Exception $exception) { $exception->fire(); }
 89: 
 90:         // determine which color to use
 91:         assert('array_key_exists($type, $this->counters); // log file type is not set yet!');
 92:         $color = ($this->counters[$type] % 2 == 0) ? 'green' : 'blue';
 93: 
 94:         // open file for writing
 95:         try {
 96:             if ((!$logFile = fopen(BASEPATH . self::LOG_DIRECTORY . '/' . "{$type}.log", 'a'))) {
 97:                 throw new Fari_Exception('Failed to open log file for writing '
 98:                     . BASEPATH . self::LOG_DIRECTORY . '/' . "{$type}.log");
 99:             }
100:             // append to file...
101:             fwrite($logFile, $this->formatTime($color) . "\n  {$this->rLogHighlight($content)}\n\n");
102:             // close file
103:             fclose($logFile);
104:         } catch (Fari_Exception $exception) { $exception->fire(); }
105:     }
106: 
107:     /**
108:      * Will format & highlight timestamp
109:      * @param string $color to apply
110:      * @return string
111:      */
112:     private function formatTime($color) {
113:         $time = explode(' ', date(DATE_RSS, mktime()));
114:         $time[4] = $this->rLogHighlight($time[4], $color);
115:         return implode(' ', $time);
116:     }
117: 
118:     /**
119:      * RLog highlighting function
120:      * @param string $string to highlight
121:      * @param string $color to apply
122:      * @param boolean $underline apply?
123:      */
124:     private function rLogHighlight($string, $color='black', $underline=FALSE) {
125:         // apply underline?
126:         $underline = ($underline) ? '4' : '1';
127: 
128:         // color switcher
129:         switch ($color) {
130:             case "magenta":
131:                 return "[{$underline};36;1m{$string}[0m";
132:             case "violet":
133:                 return "[{$underline};35;1m{$string}[0m";
134:             case "blue":
135:                 return "[{$underline};34;1m{$string}[0m";
136:             case "yellow":
137:                 return "[{$underline};33;1m{$string}[0m";
138:             case "green":
139:                 return "[{$underline};32;1m{$string}[0m";
140:             case "red":
141:                 return "[{$underline};31;1m{$string}[0m";
142:             case "gray":
143:                 return "[{$underline};30;1m{$string}[0m";
144:             case "black":
145:             default:
146:                 return "[{$underline};29;1m{$string}[0m";
147:         }
148:     }

 20: final class Fari_ApplicationRoute extends Fari_Bag {

 22:     /**#@+ $_GET variable containing route (coming from .htaccess) */
 23:     const ROUTE = 'route';
 24:     /**#@-*/
 25: 
 26:     /**#@+ default action and error presenter */
 27:     const DEFAULT_ACTION = 'actionIndex';
 28:     const ERROR_PRESENTER = 'Error404';
 29:     /**#@-*/
 30: 
 31:     /**
 32:      * Parses the request and forms route components.
 33:      */
 34:     public function __construct() {
 35:         $request = @$_GET[self::ROUTE];
 36:         if (!empty($request)) $request = $this->checkRoutes($request);
 37:         $this->route = $this->cleanup(explode('/', $request));
 38: 
 39:         if (empty($this->route[0])) {
 40:             $this->setDefaultPresenter();
 41:             $this->setDefaultAction();
 42:         } else if (empty($this->route[1])) {
 43:             $this->presenter = ucwords($this->route[0]) . 'Presenter';
 44:             $this->setDefaultAction();
 45:         } else {
 46:             $this->presenter = ucwords($this->route[0]) . 'Presenter';
 47:             $this->action = 'action' . ucwords($this->route[1]);
 48:             
 49:             $this->parameters = array_slice($this->route, 2);
 50:         }
 51:     }
 52: 
 53:     /**
 54:      * Default presenter set for the application.
 55:      */
 56:     public function setDefaultPresenter() {
 57:         $this->presenter = DEFAULT_PRESENTER . 'Presenter';
 58:     }
 59: 
 60:     /**
 61:      * Default action (actionIndex).
 62:      */
 63:     public function setDefaultAction() {
 64:         $this->action = self::DEFAULT_ACTION;
 65:     }

 68:      * Error Presenter & action setting.
 69:      */

 75:     /**
 76:      * Check if we have the routes config file and if so include it and check for routes matches on the request.
 77:      * @param string $request Original unsplit to check for match in custom routes
 78:      * @return string Optionally updated route with a custom one
 79:      */
 80:     private function checkRoutes($request) {
 81:         // form the path to the routes file
 82:         $routesFile = BASEPATH . '/config/routes' . EXT;
 83:         // do we have the routes file?
 84:         if (is_readable($routesFile)) {
 85:             // include, now we have the $customRoutes array
 86:             include($routesFile);
 87:             // if we have $customRoutes to traverse
 88:             if (is_array($customRoutes)) {
 89:                 // traverse routes
 90:                 foreach ($customRoutes as $pattern => $result) {
 91:                     // we have a match
 92:                     if (preg_match($pattern, $request)) {
 93:                         // 'change' the request by replacing the pattern match
 94:                         $request = preg_replace($pattern, $result, $request);
 95:                         // don't bother with more routes, first come first served basis
 96:                         break;
 97:                     }
 98:                 }
 99:             }
100:         }
101:         return $request;
102:     }
103: 
104:     /**
105:      * Sanitize input route array.
106:      * @param array $route
107:      * @return array
108:      */
109:     private function cleanup(array $route) {
110:         // cleanup URL to only include the following:
111:         foreach ($route as &$part) {
112:             // Fari_Filter::url()
113:             $part = filter_var($part, FILTER_SANITIZE_ENCODED);
114:         }
115:         return $route;
116:     }

 15:  * Tiny YAML parser.
 16:  *
 17:  * @copyright Copyright (c) 2008, 2010 Radek Stepan
 18:  * @package   Fari Framework\File
 19:  */
 20: class Fari_FileYaml {
 21: 
 22:     /** @var result public so we can access data repeatedly */
 23:     public $result = array();
 24: 
 25:     /**
 26:      * Parses yml file into an array or throws an exception.
 27:      * @param string $file filename, basepath will be prepended
 28:      * @return array or FileNotFoundException
 29:      */
 30:     public function toArray($file) {
 31:         assert('!empty($file); // filename cannot be empty');
 32:         // prepend slash?
 33:         $file = (substr($file, 0, 1) !== '/') ? "/{$file}" : $file;
 34:         
 35:         // read in yml file
 36:         assert('defined(\'BASEPATH\'); // base path not defined');
 37:         if (file_exists($f = BASEPATH . $file)) {
 38:             // read in file and explode on newlines
 39:             $content = explode("\n", file_get_contents($f));
 40:             $size = count($content);
 41:             // traverse
 42:             for ($i=0; $i < $size; $i++) {
 43:                 // a comment line? skip...
 44:                 if ($this->isComment($content[$i])) continue;
 45: 
 46:                 // parent?
 47:                 if ($this->isParent($content[$i])) {
 48:                     // whitespace go away
 49:                     $parent = preg_replace('/ /', '', substr($content[$i], 0, -1));
 50:                     // add new parent element
 51:                     $this->result[$parent] = array();
 52:                 // ... or child?
 53:                 } else {
 54:                     // format child element
 55:                     $element = $this->formatNode($content[$i]);
 56:                     // add to result set
 57:                     $this->result[$parent][$element[0]] = $element[1];
 58:                 }
 59:             }
 60: 
 61:             // return this baby
 62:             return $this->result;
 63:         } else {
 64:             throw new FileNotFoundException();
 65:         }
 66:     }
 67: 
 68:     /**
 69:      * Format node trimming excessive whitespace and quotation marks
 70:      * @example hello: "Hello world" => array(hello => "Hello world")
 71:      * @param string $node
 72:      * @return array
 73:      */
 74:     private function formatNode($node) {
 75:         assert('strpos($node, ":") !== FALSE; // node not in proper format');
 76:         $divider = strpos($node, ':');
 77:         return array(
 78:             // give us whatever is before colon and trim whitespace
 79:             trim(substr($node, 0, $divider)),
 80:             // give us whatever is after semicolon, trip whitespace and double quotation marks
 81:             trim(preg_replace("/\"/", '', substr($node, $divider + 1)))
 82:         );
 83:     }
 84: 
 85:     /**
 86:      * Determine if node is a parent.
 87:      * @param string $node
 88:      * @return true if node is a parent
 89:      */
 90:     private function isParent($node) {
 91:         assert('strpos($node, ":") !== FALSE; // node not in proper format');
 92:         // parent is terminated by colon
 93:         return (substr($node, -1, 1) == ':');
 94:     }
 95: 
 96:     /**
 97:      * Is this a comment line?.
 98:      * @param string $node
 99:      * @return true if node is a comment line
100:      */
101:     private function isComment($node) {
102:         return (strpos(trim($node), 0, 1) == '#');
103:     }

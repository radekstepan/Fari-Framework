37:      */
38:     public function dbToXML($parameters) {
39:         // dom string
40:         $DOMDocument = new DOMDocument('1.0', 'UTF-8');
41: 
42:         // fetch the data array
43:         $items = $this->items($parameters);
44: 
45:         // check we actually have an array
46:         try { if (!is_array($items)) {
47:             throw new Fari_Exception('Fari_Backup expects an array of items.'); }
48:         } catch (Fari_Exception $exception) { $exception->fire(); }
49:                 
50:         // <table> root element
51:         $table = $DOMDocument->appendChild($DOMDocument->createElement('table'));
52:         
53:         // traverse through all records
54:         foreach ($items as $item) {
55:             // get array keys of the item
56:             // we could explode $columns as well if they are passed
57:             $keys = array_keys($item);
58:                         
59:             // <table><row> elemenent we will always have
60:             $row = $table->appendChild($DOMDocument->createElement('row'));
61:                         
62:             // traverse through keys/columns
63:             foreach ($keys as $column) {
64:                 // <table><row><column> value, escaped
65:                 $row->appendChild($DOMDocument->createElement($column, Fari_Escape::xml($item[$column])));
66:             }
67:         }
68:                 
69:         // generate XML and return
70:         $DOMDocument->formatOutput = TRUE;
71:         return $DOMDocument->saveXML();
72:     }

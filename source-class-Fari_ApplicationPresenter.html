 37:     /**
 38:      * Makes for nicer method calls.
 39:      * @param string $method
 40:      * @param mixed $params
 41:      * @throws Fari_Exception
 42:      */
 43:     public function __call($method, $params) {
 44:         try {
 45:             // determine the method called
 46:             if (!preg_match('/^(render|send)(\w+)$/', $method, $matches)) {
 47:                 throw new Fari_Exception("Call to undefined method {$method}");
 48:             }
 49: 
 50:             // what do you want?
 51:             switch ($matches[1]) {
 52:                 // render something with a passed type
 53:                 case 'send':
 54:                 case 'render':
 55:                     // pass on var number of params
 56:                     if (count($params) > 1) {
 57:                         $this->render($params, $matches[2]);
 58:                     } else {
 59:                         $this->render(current($params), $matches[2]);
 60:                     }
 61:                     break;
 62:                 default:
 63:                     throw new Fari_Exception("Call to undefined parameter {$matches[1]}");
 64:             }
 65: 
 66:         } catch (Fari_Exception $exception) { $exception->fire(); }
 67:     }

130:     /**
131:      * Tries to call your custom render action and then displays the view template.
132:      * @param string $viewName
133:      * @param mixed $parametres Parameters to pass to the function
134:      */
135:     function renderAction($viewName=NULL, $parameters=NULL) {
136:         // check we are calling the proper method
137:         try {
138:             if (!is_null($viewName) && strpos($viewName, '/') !== FALSE) {
139:                 throw new Fari_Exception('Use renderTemplate if you want to call a custom template.');
140:             }
141:         } catch (Fari_Exception $exception) { $exception->fire(); }
142: 
143:         // define view presenter prefix and action/template name
144:         list ($presenter, $action) = $this->getPresenterAction($viewName);
145:         
146:         // do we have a render function to call?
147:         // It is always called in our presenter!
148:         if (method_exists($this->request->getPresenter(), 'render' . $action)) {
149:             // reflect on the method
150:             $method = new Fari_ApplicationReflection($this, 'render' . ucfirst($action));
151:             
152:             // set parameters
153:             if ($method->hasParameters()) $method->setParameters(&$parameters);
154:             // call the render method
155:             $method->call($this);
156:         }
157: 
158:         // strip prefix Presenter suffix :)
159:         if (($suffixStart = strpos($presenter, 'Presenter')) !== FALSE) {
160:             $presenter = ucfirst(substr($action, 0, $suffixStart + 1));
161:         }
162: 
163:         // render file through a view
164:         $this->response->renderView("{$presenter}/{$action}", &$this->bag->values);
165: 
166:         // we don't want any further code to be executed in the Presenter
167:         die();
168:     }

223:     /**
224:      * Get prefix prefix and action/template name from view name.
225:      * @param string $viewName
226:      * @return array
227:      */
228:     private function getPresenterAction($viewName=NULL) {
229:         // if view is not specified get the action that was called...
230:         if (!isset($viewName)) {
231:             // ... and strip action prefix, then lowercase
232:             $viewName = strtolower(substr($this->request->getAction(), 6));
233:         }
234: 
235:         // get the prefix of the view if present
236:         if (($slash = stripos($viewName, '/')) !== FALSE) {
237:             // the prefix Presenter named directory where our view resides
238:             $prefix = ucwords(substr($viewName, 0, $slash));
239:             // the actual name of the view we have requested
240:             $viewName = substr($viewName, $slash + 1);
241:         } else {
242:             // set default prefix as this presenter
243:             $presenter = $this->request->getPresenter();
244:             $prefix = substr($presenter, 0, stripos($presenter, "Presenter"));
245:         }
246: 
247:         // form an array pair
248:         return array($prefix, $viewName);
249:     }

273:     /**
274:      * Will try to load filters.
275:      * @param string $filtersVar name of variable storing filters.
276:      */
277:     private function loadFilters($filtersVar) {
278:         assert('is_string($filtersVar); // pass variable as a string');
279:         // now for custom filters... these will throw exception if not found
280:         // do we have them at all?
281:         if (isset($this->$filtersVar)) {
282:             foreach ($this->$filtersVar as $filter) {
283:                 // check we are properly formatted
284:                 $filterName = current(array_keys($filter));
285:                 $applyBefore = $filter[$filterName];
286:                 
287:                 // precondition
288:                 assert('!empty($filterName) || !is_array($applyBefore); // malformed filter format, use
289:                     "array(\'filterName\' => array(\'new\'))"');
290: 
291:                 // determine action called by trimming off "action" prefix
292:                 $actionCalled = strtolower(substr($this->request->getAction(), 6));
293: 
294:                 // are we applying the filter?
295:                 if (in_array($actionCalled, $applyBefore)) {
296:                     // oh yeah
297:                     $this->applyFilter('filter' . ucfirst($filterName));
298:                 }
299:             }
300:         }
301:     }
302: 
303:     /**
304:      * Call filter in our presenter.
305:      * @param string $filterMethod
306:      */
307:     private function applyFilter($filterMethod) {
308:         try {
309:             if (method_exists($this->request->getPresenter(), $filterMethod)) {
310:                 // call the method
311:                 $this->$filterMethod();
312:             } else {
313:                 // else... exception land baby
314:                 throw new Fari_Exception("Filter method '{$filterMethod}' not found.");
315:             }
316:         } catch (Fari_Exception $exception) { $exception->fire(); }
317:     }

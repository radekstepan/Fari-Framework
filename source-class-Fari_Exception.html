104:     /**
105:      * Format mixed variables for output
106:      * @param <type> $mixed
107:      * @return <type>
108:      */
109:     public static function formatVars($mixed) {
110:         // we are working in HTML context
111:         //$mixed = Fari_Escape::html($mixed);
112:         if ($mixed === NULL) $mixed = '<em>NULL</em>';
113:         else if (empty($mixed)) $mixed = '<em>empty</em>';
114:         else if (is_string($mixed)) $mixed = Fari_Escape::html($mixed);
115:         else {
116:             ob_start();
117:             var_dump($mixed);
118:             $mixed = ob_get_contents();
119:             ob_clean();
120: 
121:             $mixed = explode("\n", $mixed);
122:             foreach ($mixed as &$line) {
123:                 // how big is the whitespace on the left?
124:                 $padding = strlen($line) - strlen(ltrim($line));
125:                 // add extra padding for better readability
126:                 for ($i=0;$i<$padding;$i++) $line = "  " . $line;
127:                 // if our line contains a value give it extra pad
128:                 if (strpos($trimmed = ltrim($line), "[") !== FALSE) {
129:                     // highlight array key
130:                     $line = str_replace("[", "<strong>[", $line);
131:                     $line = str_replace("]", "]</strong>", $line);
132:                 } else {
133:                     if (substr(trim($line), 0) != "}") $line = "   " . $line;
134:                 }
135:                 $line = substr($line, 3);
136:             }
137:             $mixed = implode("\n", $mixed);
138:         }
139: 
140:         return $mixed;
141:     }

232:     /**
233:      * Output a syntax highlighted sourcecode.
234:      * @param string $errorFile Is the file where exception was thrown
235:      * @param string $errorLine Line we want to highlight as troublesome
236:      * @param string $displayRange How many source code lines before and after error line to show
237:      * @param string $divId So that we can show/hide some source and then call it via js
238:      */
239:     static function showErrorSource($errorFile, $errorLine, $displayRange=6, $divId=0) {
240:         // set color of the highlighting
241:         ini_set('highlight.string', '#080');
242:         ini_set('highlight.comment',    '#999; font-style: italic');
243:         ini_set('highlight.default',    '#33393c');
244:         ini_set('highlight.html',   '#06b');
245:         ini_set('highlight.keyword',    '#d24; font-weight: bold');
246: 
247:         // get the source code into a string
248:         $sourceCode = highlight_file($errorFile, TRUE);
249:         // split into an array so that we can extract lines
250:         $sourceCode = explode('<br />', $sourceCode);
251: 
252:         // where (which line) to start showing source code? (start at line 1 ;)
253:         $beginLine = max(1, $errorLine - $displayRange);
254:         // where to stop?
255:         $endLine = min($errorLine + $displayRange, count($sourceCode));
256: 
257:         // open div with the error message
258:         if ($divId != '0') echo '<div id="' . $divId . '" class="code" style="display:none;">';
259:         else echo '<div id="' . $divId . '" class="code">';
260: 
261:         // highlighting might have started before we 'cut' it
262:         // set pointer to the beginning of our output
263:         $pointer = $beginLine;
264:         // while we haven't reached the start of file...
265:         while ($pointer-- > 0) {
266:             /**
267:              * Match unlimited times any character that is not a \n
268:              * Capture into backreference </span> OR <span *>.
269:              */
270:             if (preg_match('%.*(</?span[^\>]*>)%', $sourceCode[$pointer], $match)) {
271:                 // echo the highlighting tag if we've started it and not closed it
272:                 if ($match[1] !== '</span>') echo $match[1];
273:                 break;
274:             }
275:         }
276: 
277:         // paint the code
278:         // set pointer to the beginning of our output
279:         $pointer = $beginLine-1;
280:         // while we haven't reached the end of output...
281:         while (++$pointer <= $endLine) {
282:             // take our line from the source
283:             $line = $sourceCode[$pointer-1];
284:             // highlight our error line
285:             if ($pointer == $errorLine) {
286:                 // strip formatting
287:                 $line = strip_tags($line);
288:                 // add tags
289:                 echo '<h1 class="error"><span class="num err">' . $pointer . ':</span> &nbsp;&nbsp;&nbsp; ' . $line . '</h1>';
290:             // and output sourcecode line with delimiter
291:             } else echo '<span class="num">' . $pointer . ':</span> &nbsp;&nbsp;&nbsp; ' . $line . "<br />\n";
292:         }
293: 
294:         // close div
295:         echo '</div>';
296:     }

307:         // traverse the array

312:             if (isset($line)) echo '&nbsp;&nbsp;(' . $line . ')';
313:             if (isset($function)) echo '&nbsp;&nbsp;' . $function . '()&nbsp;&nbsp;';
314: 
315:             // link to a javascript function that shows/hides the code listing
316:             echo '<a href="" onclick="toggle(\'' . $counter . '\');return false;" >source</a>';
317:             // add sourcecode listing
318:             self::showErrorSource($file, $line, 6, $counter);
319:             $counter++; // add to the counter

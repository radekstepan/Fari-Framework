568:     /**
569:      * A simple table JOIN.
570:      * @param mixed $table string name we want to join to or a Table object
571:      * @param mixed $on array of left => right column or a string if column is shared in both tables
572:      * @return Table so we can call method (and add where clause)
573:      */
574:     public function join($table, $on) {
575:         // fetch table name from an object
576:         if ($table instanceof Table) $table = $table->table;
577: 
578:         // joining columns have different names
579:         if (is_array($on)) {
580:             foreach ($on as $left => $right) {
581:                 $leftTable = $this->tableName;
582:                 $rightTable = $table;
583: 
584:                 // are we passing a table name?
585:                 if (strpos($left, '.') !== FALSE) {
586:                     list($leftTable, $left) = explode('.', $left);
587:                 }
588:                 if (strpos($right, '.') !== FALSE) {
589:                     list($rightTable, $right) = explode('.', $right);
590:                 }
591: 
592:                 // filter out extra spaces
593:                 $left = preg_replace('/\s\s+]/', '', $left);
594:                 $right = preg_replace('/\s\s+]/', '', $right);
595: 
596:                 $join[] = "{$leftTable}.{$left}={$rightTable}.{$right}";
597:             }
598:             // implode the join ON params with a separator
599:             $join = implode(' AND ', $join);
600:             // create query
601:             $this->join .= " JOIN {$table} ON {$join}";
602:         // column names the same in both tables
603:         } else {
604:             try { if (strpos($on, '.') !== FALSE) {
605:                 throw new Fari_Exception('Are you trying to specify a table name? Use an array() instead.'); }
606:             } catch (Fari_Exception $exception) { $exception->fire(); }
607: 
608:             // filter out extra spaces
609:             $on = preg_replace('/\s\s+]/', '', $on);
610: 
611:             $this->join .= " JOIN {$table} ON {$this->tableName}.{$on}={$table}.{$on}";
612:         }
613: 
614:         return $this;
615:     }

698:     /**
699:      * Prepare a where clause.
700:      * @return <type>
701:      */
702:     private function prepareWhere() {
703:         $result = '';
704:         $i = 0;
705:         foreach ($this->where as $column => $value) {
706:             // WHERE (NOT) IN ()
707:             if (strpos($operator = $this->findOperator($value), 'IN') !== FALSE) {
708:                 // no binding occurs...
709:                 $result .= "{$column} {$value} AND ";
710:             // the rest...
711:             } else {
712:                 $result .= "{$column} {$operator} :id{$i} AND ";
713:                 $i++;
714:             }
715:         }
716: 
717:         return substr($result, 0, -5);
718:     }

722:      * @param <type> $statement
723:      * @return <type>
724:      */
725:     private function bindWhere($statement) {
726:         $i = 0;
727:         foreach ($this->where as $column => &$value) {

 14: /**
 15:  * Caching a view to a file and fetching it back again.
 16:  *
 17:  * @copyright Copyright (c) 2008, 2010 Radek Stepan
 18:  * @package   Fari Framework\Application
 19:  */
 20: final class Fari_ApplicationViewCache {
 21:     
 22:     /**#@+ cache files directory & extension */
 23:     const CACHE_DIR = '/tmp/';
 24:     const CACHE_EXT = '.html';
 25:     /**#@-*/
 26: 
 27:     /**#@+ view file suffix */
 28:     const VIEW_SUFFIX = '.phtml';
 29:     /**#@-*/
 30: 
 31:     /** @var cache expiry in minutes */
 32:     private $expiry = 5;
 33: 
 34:     /** @var fileId used when caching a file */
 35:     private $fileId;
 36: 
 37:     /** @var view name to cache */
 38:     private $viewName;
 39: 
 40:     /**
 41:      * We want to use a view to render our data?
 42:      * @param string $viewName
 43:      * @param mixed $parameters Parameters to pass to the function that identify the view
 44:      */
 45:     public function renderCache($viewName, $parameters=NULL) {
 46:         assert('strlen($viewName) > 0; // view name cannot be empty');
 47: 
 48:         $this->viewName = $viewName;
 49:         // build file id from (folder(s) and) file name
 50:         $this->fileId = $this->getFileId($viewName, $parameters);
 51: 
 52:         // check if dir writable
 53:         $this->isDirWritable(self::CACHE_DIR);
 54: 
 55:         // everything went OK, 'build' cache filename path
 56:         $cacheFile = BASEPATH . self::CACHE_DIR . $this->fileId . self::CACHE_EXT;
 57: 
 58:         // serve the cached file if it exists and is not too old
 59:         if (file_exists($cacheFile) && !$this->isCacheOld($cacheFile)) {
 60:             readfile($cacheFile);
 61:             die();
 62:         }
 63:     }
 64: 
 65:     /**
 66:      * Save a view into a cache and dump it
 67:      * @param array values to extract into the file
 68:      */
 69:     public function saveCache(array $values) {
 70:         // include view helpers
 71:         include BASEPATH . "/fari/Application/ApplicationViewHelpers" . EXT;
 72: 
 73:         // we need to have the fileId and name set!
 74:         try { if (!isset($this->fileId)) {
 75:             throw new Fari_Exception('Fari requires you to call renderCache() from a Presenter first.'); }
 76:         } catch (Fari_Exception $exception) { $exception->fire(); }
 77: 
 78:         // everything went OK, 'build' cache filename path
 79:         $cacheFile = BASEPATH . self::CACHE_DIR . $this->fileId . self::CACHE_EXT;
 80: 
 81:         // form views path
 82:         $path = BASEPATH . '/' . APP_DIR . '/views/';
 83:         $viewFile = $path . $this->viewName . self::VIEW_SUFFIX;
 84:         // check if view exists
 85:         $this->isViewValid($viewFile);
 86: 
 87:         // are we using a @layout file?
 88:         $temp = explode('/', $viewName);
 89:         assert('count($temp) == 2; // $viewName needs to consist of "presenter/file"');
 90:         // custom layout named after our presenter
 91:         if (file_exists($layout = $path . '@' . strtolower($temp[0]) . self::VIEW_SUFFIX)) {
 92:             $cacheFile = $this->returnLayoutAndView($layout, $viewFile);
 93:         // application level layout
 94:         } else if (file_exists($layout = $path . '@application' . self::VIEW_SUFFIX)) {
 95:             $cacheFile = $this->returnLayoutAndView($layout, $viewFile);
 96:         } else {
 97:             // import key:value array into symbol table
 98:             extract($values, EXTR_SKIP);
 99: 
100:             // no layout...
101:             ob_start();
102:             include $viewFile;
103:             $cacheFile = ob_get_contents();
104:             ob_end_clean();
105:         }
106: 
107:         // write into cache file
108:         $this->writeCache($contentOutput, $cacheFile);
109: 
110:         // send the output to the browser
111:         echo $cacheFile;
112:     }
113: 
114:     /**
115:      * Build file id as file might be in a sub directory etc.
116:      * @param string $viewName View name that we are working with
117:      * @param string $extraParam Extra parameter to find the view by
118:      * @return string MD5 hashed view file id
119:      */
120:     private function getFileId($viewName, $extraParam) {
121:         // calculate based on md5
122:         if (isset($extraParam)) return md5($viewName.$extraParam);
123:         return md5($viewName);
124:     }
125:     
126:     /**
127:      * Write contents of a view/template to a file.
128:      * @param string $contentOutput Content of the view we want to write
129:      * @param string $cacheFile Filename we will write to
130:      */
131:     private function writeCache($contentOutput, $cacheFile) {
132:         // open file for writing
133:         $cacheFile = fopen($cacheFile, 'w');
134:         // write file
135:         fwrite($cacheFile, $contentOutput);
136:         // close file
137:         fclose($cacheFile);
138:     }
139:     
140:     /**
141:      * Will check if a cache directory is writable. Will throw Fari_Exception on error.
142:      * @param string $cacheDir Directory that stores our cached files
143:      * @return void
144:      */
145:     private function isDirWritable($cacheDir) {
146:         try {
147:             if (!is_writable(BASEPATH . $cacheDir)) {
148:                 throw new Fari_Exception('Cache directory ' . $cacheDir . ' is not writable.');
149:             }
150:         } catch (Fari_Exception $exception) { $exception->fire(); }
151:     }
152:     
153:     /**
154:      * Check if view file can be included. Will throw Fari_Exception on error.
155:      * @param string $viewFile to check for existence
156:      */
157:     private function isViewValid($viewFile) {
158:         try {
159:             // check if file path exists
160:             if (!file_exists($viewFile)) {
161:                 throw new Fari_Exception('View not located in: ' . $viewFile);
162:             }
163:         } catch (Fari_Exception $exception) { $exception->fire(); }
164:     }
165:     
166:     /**
167:      * Determines whether cache file is too old to be used.
168:      * @param string $cacheFile Filename to check for 'freshness'
169:      * @return boolean Whether we need to get a fresh copy (TRUE) or not (FALSE)
170:      */
171:     private function isCacheOld($cacheFile) {
172:         // time now - cache lifetime
173:         $goodTime = time() - ($this->expiry * 60);
174:         $cacheAge = filemtime($cacheFile);
175:         // return TRUE if file age 'within' good time
176:         return ($goodTime < $cacheAge) ? FALSE : TRUE;
177:     }
178: 
179:     /**
180:      * Return layout file and view file included in $template var.
181:      * @param string $layout file path
182:      * @param string $view file path
183:      * @return string cached file
184:      */
185:     private function returnLayoutAndView($layout, $view) {
186:         // import key:value array into symbol table
187:         extract($values, EXTR_SKIP);
188: 
189:         ob_start();
190:         // save view into template var
191:         include $view;
192:         $template = ob_get_contents();
193:         ob_end_clean();
194: 
195:         ob_start();
196:         // call parent layout
197:         include $layout;
198:         $cacheFile = ob_get_contents();
199:         ob_end_clean();
200: 
201:         return $cacheFile;
202:     }
